### <span className="en">TypeScript Type Definitions</span><span className="zh">TypeScript 型別定義</span>

<p><span className="en">Copy these type definitions into your project for type-safe token usage. Autocomplete and compile-time checking prevent off-brand color/spacing usage.</span><span className="zh">將這些型別定義複製到您的專案中，實現型別安全的代符使用。自動完成和編譯時檢查可防止不符品牌的色彩/間距使用。</span></p>

<CodeBlock>{`// gps-tokens.d.ts — GPS Brand Bible Design Token Types

export type GPSColorPrimary = 'turquoise' | 'copper';
export type GPSColorSecondary = 'purple' | 'green' | 'orange';
export type GPSColorNeutral = 'black' | 'ink' | 'charcoal' | 'slate'
  | 'g500' | 'g400' | 'g300' | 'g200' | 'g100' | 'g50' | 'white';
export type GPSColorSemantic = 'success' | 'warning' | 'danger';

export type GPSColorScale = 50 | 100 | 200 | 300 | 400 | 500
  | 600 | 700 | 800 | 900 | 950;

export type GPSSpacingToken = 'sp1' | 'sp2' | 'sp3' | 'sp4'
  | 'sp6' | 'sp8' | 'sp12' | 'sp16' | 'sp24';

export type GPSBreakpoint = 'sm' | 'md' | 'lg' | 'xl';

export type GPSFontFamily = 'en' | 'zh' | 'display' | 'mono';

export type GPSRadius = 'sm' | 'md' | 'lg' | 'xl' | 'full';

export type GPSMotionEasing = 'reveal' | 'settle' | 'standard';
export type GPSMotionDuration = 'fast' | 'normal' | 'slow';

export interface GPSTokens {
  color: {
    primary: Record<GPSColorPrimary, Record<GPSColorScale, string>>;
    secondary: Record<GPSColorSecondary, Record<GPSColorScale, string>>;
    neutral: Record<GPSColorNeutral, string>;
    semantic: Record<GPSColorSemantic, string>;
  };
  spacing: Record<GPSSpacingToken, string>;
  breakpoint: Record<GPSBreakpoint, string>;
  font: Record<GPSFontFamily, string>;
  radius: Record<GPSRadius, string>;
}`}</CodeBlock>
